query{ for cus in db.Customer do
       join sh in db.``Sales Header`` on (cus.``No.`` = sh.``Sell-to Customer No.``)
       where (cus.Name.StartsWith "Steffen") 
       select (cus.Name,sh.``No.``,sh.``Currency Code``) } 
  |> Seq.toArray
